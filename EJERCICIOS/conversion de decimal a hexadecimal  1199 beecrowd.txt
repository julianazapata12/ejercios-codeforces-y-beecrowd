# de decimal a hexadecimal
def deci_a_hexa(deci):
    hexa = ""
    while deci > 0:
        residuo = deci % 16
        verdadero = obt_hexa(residuo)
        hexa = verdadero + hexa
        deci = int(deci/ 16)
    return hexa
